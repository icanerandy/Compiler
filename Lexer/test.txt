// 三递归函数调用测试3：花式上楼梯，比如已知总共有15级台阶
//一个人从地面开始可以一次上一级台阶，也可以一次上两级台阶
//最多可以一次上三级台阶，请问，上到15级有多少中上楼梯的方法
//程序可以使用倒推的方法来写
//结果：1，2，4，7，13，24，44，81，149......
int f(int);
main()
{
   int m;
   m = read();
   write(f(m));

}

int f(int n) {
   int a;
   if(n==1) {
		return 1;
	}
   else if(n==2) {
		return 2;
	}
   else if(n==3) {
		return 4;
	}
    a = f(n - 1)+ f(n - 2)+ f(n - 3);
    return a ;
 }
